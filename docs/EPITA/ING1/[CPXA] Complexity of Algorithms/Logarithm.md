$$
ln(x) = y \iff x = e^y \\
log_2x = y \iff x = 2^y \\
log_{10}x = y \iff x = 10^y
$$

$$
log(ab) = log(a)+log(b)\\
log(a/b) = log(a) -log(b)\\
log(a^b)=blog(a)
$$

$$
log_b(y) = \frac{ln(y)}{ln(b)}
$$
